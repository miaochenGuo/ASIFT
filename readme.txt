# 这里是实现Asift 和Aorb在opencv配置下的工程
# 参考链接
http://www.mattsheckells.com/opencv-asift-c-implementation/


# 对于Asift源码的链接：

# Compile
make sure you have OpenCV and linux

cd Aorb_sift
mkdir build
cd build
cmake ..
make
cd ..

# Usage
#  example
build/Aorb_demo
build/Aorb_demo
